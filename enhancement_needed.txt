# Continuous monitoring script
import requests
import time
import json

def monitor_arbitrage():
    base_url = "http://localhost:8000/api"
    
    while True:
        try:
            # Get current opportunities
            response = requests.get(f"{base_url}/opportunities/")
            data = response.json()
            
            if data['count'] > 0:
                best_opp = data['opportunities'][0]
                print(f"[{time.strftime('%H:%M:%S')}] Best opportunity: {best_opp['profit_percentage']:.4f}%")
                
                # Auto-execute if profit > threshold
                if best_opp['profit_percentage'] > 0.5:  # 0.5% threshold
                    trade_data = {
                        "triangle": best_opp['triangle'],
                        "amount": 100,
                        "exchange": "binance"
                    }
                    trade_response = requests.post(f"{base_url}/trading/execute/", json=trade_data)
                    print(f"Executed trade: {trade_response.json()}")
            
            time.sleep(5)  # Check every 5 seconds
            
        except Exception as e:
            print(f"Monitoring error: {e}")
            time.sleep(10)
